int** createMatrix(int n){
	int** retVal = new int*[n];

	for(int i = 0; i < n; i++){
		retVal[i] = new int[n];
	}
	return retVal;
}

//Liberar memoria
void freeMatrix(int n, int** matrix){
	for(int i = 0; i < n;i++){
		delete[] matrix[i];
	}
	delete[] matrix;

}
